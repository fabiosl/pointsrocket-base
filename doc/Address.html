def moip_attributes
  attrs = ActiveSupport::HashWithIndifferentAccess.new(self.attributes)
  merge = {}
  if self.zipcode
    merge = {
      zipcode: self.zipcode.gsub('-', '')
    }
  end
  attrs.slice(:street, :number, :district, :state, :country, :city, :complement).merge(merge)
end
