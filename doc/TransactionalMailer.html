def contactation_created(contact)
  mandrill_mail(
    template: 'contactation-created',
    to: @admins,
    important: true,
    inline_css: true,
    vars: {
      'CONTACT_NAME' => contact.name,
      'CONTACT_EMAIL' => contact.email,
      'CONTACT_PHONE' => contact.phone,
      'CONTACT_MESSAGE' => contact.message
    },
    recipient_vars: @admins.map do |admin|
      { admin[:email] =>
        {
          'ADMIN_NAME' => admin[:name]
        }
      }
    end
  )
end
