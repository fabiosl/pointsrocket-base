def create_vote question_or_answer, status, points
  vote_params = get_vote_params(question_or_answer)

  vote = @user.votes.create(vote_params.merge({
    status: Vote.statuses[status]
  }))

  question_or_answer.user.points.create(
    value: points,
    pointable: vote
  )
end

def destroy_vote vote, question_or_answer

    pointable: vote
  ).destroy_all

  vote.destroy
end

def flow question_or_answer, status, points
  vote_params = get_vote_params(question_or_answer)

  vote = @user.votes.where(vote_params).first

  if vote
    if vote.status.to_sym != status
      self.destroy_vote vote, question_or_answer
      self.create_vote question_or_answer, status, points
    else
      self.destroy_vote vote, question_or_answer
    end
  else
    self.create_vote(question_or_answer, status, points)
  end

end

def get_vote_params question_or_answer
  if question_or_answer.is_a? Question
    {question: question_or_answer}
  else
    {answer: question_or_answer}
  end
end
